    要编写一个读取PDF文件内容的Python程序，我们可以使用PyPDF2库。首先，确保你已经安装了这个库。如果没有安装，可以使用以下命令安装：

    bash
    pip install PyPDF2

    然后，可以使用以下代码读取PDF文件的内容：

    python
    import PyPDF2

# 打开PDF文件
with open('example.pdf', 'rb') as file:
    # 创建一个PDF阅读器对象
    pdf_reader = PyPDF2.PdfFileReader(file)

    # 获取PDF的页数
    num_pages = pdf_reader.numPages
    print(f"Total number of pages: {num_pages}")

    # 读取并打印每一页的内容
    for page_num in range(num_pages):
        # 获取当前页
        page = pdf_reader.getPage(page_num)

        # 读取当前页的文本内容
        page_text = page.extractText()

        # 打印当前页的内容
        print(f"Page {page_num + 1}:\n{page_text}\n{'-' * 40}\n")

    请将上述代码中的example.pdf替换为你要读取的PDF文件名。这个程序将输出PDF文件中每一页的内容。

    为了根据提取到的最大金额值重命名PDF文件，我们可以在循环中添加重命名操作。以下是修改后的代码：

    python
    import re
import os
import pdfplumber

def read_pdf_content(file_path):
    with pdfplumber.open(file_path) as pdf:
        full_text = "".join(page.extract_text() for page in pdf.pages)
    return full_text

def get_pdf_files(pdf_dir):
    pdf_files = []
    for root, dirs, files in os.walk(pdf_dir):
        for file in files:
            if file.endswith(".pdf"):
                pdf_files.append(os.path.join(root, file))
    return pdf_files

def rename_pdf_file(file_path, new_value):
    dir_path = os.path.dirname(file_path)
    new_file_name = f"发票{new_value}.pdf"
    new_file_path = os.path.join(dir_path, new_file_name)
    os.rename(file_path, new_file_path)
    return new_file_path

pdf_dir = r"E:\Dropbox\发票\4月份"
pdf_files = get_pdf_files(pdf_dir)
print(pdf_files)

values = []
renamed_pdf_files = []
for pdf_file in pdf_files:
    pdf_text = read_pdf_content(pdf_file)
    pattern = r'[¥￥](\d+(?:\.\d{1,2})?)'
    result = re.findall(pattern, pdf_text)
    print(result)

    if result:
        max_value = max([float(i) for i in result])
        values.append(max_value)
        new_pdf_file = rename_pdf_file(pdf_file, max_value)
        renamed_pdf_files.append(new_pdf_file)

print(values)
print(renamed_pdf_files)
